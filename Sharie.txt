sharie:
-help users share images, videos like ig and 

entities/models:
- users:
	{
  _id: ObjectId,
  uid: String, // For social auth (Google/Facebook/Apple), optional
  username: String,
  email: String,
  name: String,
  bio: String,
  avatar: String, // URL
  followers: [ObjectId], // userIds
  following: [ObjectId],
  postCount: Number, // optional
  createdAt: Date,
  updatedAt: Date
}

- post:
	{
  _id: ObjectId,
  author: ObjectId, // ref to User
  caption: String,
  media: [
    {
      url: String,
      type: "image" | "video"
    }
  ],
  likes: [ObjectId], // userIds
  shares: [ObjectId], // userIds
  commentCount: Number,
  likeCount: Number,
  createdAt: Date,
  updatedAt: Date
}

- commment:{
  _id: ObjectId,
  postId: ObjectId, // ref to Post
  author: ObjectId, // ref to User
  text: String,
  parentCommentId: ObjectId | null, // For replies
  createdAt: Date
}

	
- message:
	{
  _id: ObjectId,
  chatId: ObjectId, // ref to Chat
  sender: ObjectId, // ref to User
  text: String,
  media: {
    url: String,
    type: "image" | "video"
  } | null,
  seenBy: [ObjectId], // userIds
  createdAt: Date
}

- chat:	
	{
  _id: ObjectId,
  participants: [ObjectId], // userIds
  lastMessage: {
    text: String,
    sender: ObjectId,
    createdAt: Date
  },
  updatedAt: Date,
  isGroup: Boolean // for future
}

- notification:
	{
  _id: ObjectId,
  userId: ObjectId, // who receives the notification
  sourceUserId: ObjectId, // who triggered it
  type: "like" | "comment" | "follow" | "message",
  postId: ObjectId | null,
  isRead: Boolean,
  createdAt: Date
}
